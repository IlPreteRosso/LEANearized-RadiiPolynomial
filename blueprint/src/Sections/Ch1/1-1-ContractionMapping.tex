\subsection{Contraction Mapping Theorem (Section 2.1)}\label{sec:contraction}

\begin{definition}[Complete metric space]\label{def:CM_Complete}
  \lean{RP.CM_Complete}
  Let $(X,\mathbf d)$ be a metric space. We say $X$ is \emph{complete} if every Cauchy sequence
  converges in $X$.
\end{definition}

\begin{definition}[Contraction]\label{def:CM_Contraction}
  \lean{RP.CM_Contraction}
  Let $(X,\mathbf d)$ be a metric space and let $\kappa\in[0,1)$. A map $T:X\to X$ is a
  \emph{contraction with constant $\kappa$} if for all $x,y\in X$,
  \[
    \mathbf d(Tx,Ty)\le \kappa\,\mathbf d(x,y).
  \]
\end{definition}

\begin{definition}[Picard iterates]\label{def:CM_PicardIter}
  \lean{RP.CM_PicardIter}
  Given $x_0\in X$, define $x_{n+1}=T(x_n)$, i.e.\ $x_n=T^{\,n}(x_0)$.
\end{definition}

\begin{theorem}[Contraction Mapping—existence and uniqueness]\label{thm:CM_existsUnique}
  \lean{ContractingWith.exists_fixedPoint}
  \uses{def:CM_Complete,def:CM_Contraction,def:CM_PicardIter}
  Let $(X,\mathbf d)$ be complete and $T:X\to X$ a contraction with constant $\kappa<1$.
  Then there exists a unique fixed point $\tilde x\in X$ with $T(\tilde x)=\tilde x$.
\end{theorem}


\begin{theorem}[Convergence and geometric rate]\label{thm:CM_rate}
  \lean{ContractingWith.apriori_edist_iterate_efixedPoint_le}
  \uses{thm:CM_existsUnique}
  Under the same hypotheses, for any $x\in X$ and all $n\ge 0$,
  \[
    \mathbf d\!\big(T^{\,n}(x),\tilde x\big)
      \;\le\; \frac{\kappa^{\,n}}{1-\kappa}\,\mathbf d\!\big(T(x),x\big),
  \]
  hence $T^{\,n}(x)\to\tilde x$ as $n\to\infty$.
\end{theorem}
